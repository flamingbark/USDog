62 function transferFrom(address src, address dst, uint256 wad) public returns (bool) {
63 require(balanceOf[src] >= wad, "StableCoin/insufficient-balance");
64
65 if (src != msg.sender && allowance[src][msg.sender] != type(uint256).max) {
66 require(allowance[src][msg.sender] >= wad, "StableCoin/insufficient-allowance");
67 allowance[src][msg.sender] = sub(allowance[src][msg.sender], wad);

103 // --- Permit ---
104 function permit(address owner, address spender, uint256 value, uint256 deadline, uint8 v, bytes32 r, bytes32 s) external {
105 require(deadline >= block.timestamp, "StableCoin/permit-expired");
106
107 bytes32 digest = keccak256(abi.encodePacked(
108 "\x19\x01",
109 DOMAIN_SEPARATOR,
110 keccak256(abi.encode(PERMIT_TYPEHASH, owner, spender, value, nonces[owner]++, deadline))
111 ));
112
113 address recoveredAddress = ecrecover(digest, v, r, s);
114 require(recoveredAddress != address(0) && recoveredAddress == owner, "StableCoin/invalid-permit");
115
116 allowance[owner][spender] = value;
117 emit Approval(owner, spender, value);
118 }
