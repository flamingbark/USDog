149 function frob(bytes32 i, address u, address v, address w, int256 dink, int256 dart) external {
150 Urn memory urn = urns[i][u];
151 Ilk memory ilk = ilks[i];
152
153 // Urn has been previously manipulated, live urn, or a line exists
154 require(ilk.rate != 0, "Vat/ilk-not-init");
155
156 urn.ink = _addInt(urn.ink, dink);
157 urn.art = _addInt(urn.art, dart);
158 ilk.Art = _addInt(ilk.Art, dart);
159
160 int256 dtab = _mulInt(ilk.rate, dart);
161 uint256 tab = _mul(ilk.rate, urn.art);
162 debt = _addInt(debt, dtab);
163
164 // Either debt has decreased, or debt ceilings are not exceeded
165 require(either(dart <= 0, both(_mul(ilk.Art, ilk.rate) <= ilk.line, debt <= Line)), "Vat/ceiling-exceeded");
166
167 // Urn is either less risky than before, or it is safe
168 require(either(both(dart <= 0, dink >= 0), tab <= _mul(urn.ink, ilk.spot)), "Vat/not-safe");
169
170 // Urn is either more safe, or the owner consents
171 require(either(both(dart <= 0, dink >= 0), wish(u, msg.sender)), "Vat/not-allowed-u");
172
173 // Collateral src consents
174 require(either(dink <= 0, wish(v, msg.sender)), "Vat/not-allowed-v");
175
176 // Dai dst consents
177 require(either(dart >= 0, wish(w, msg.sender)), "Vat/not-allowed-w");
178
179 // Urn has no debt, or a non-dusty amount
180 require(either(urn.art == 0, tab >= ilk.dust), "Vat/dust");
181
182 gem[i][v] = _addInt(gem[i][v], -dink);
183 dai[w] = _addInt(dai[w], dtab);
184 urns[i][u] = urn;
185 ilks[i] = ilk;
186 }
