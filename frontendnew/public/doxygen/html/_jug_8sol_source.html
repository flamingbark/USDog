52 function rpow(uint256 x, uint256 n, uint256 base_) internal pure returns (uint256 z) {
53 assembly {
54 switch x case 0 {switch n case 0 {z := base_} default {z := 0}}
55 default {
56 switch mod(n, 2) case 0 { z := base_ } default { z := x }
57 let half := div(base_, 2) // for rounding.
58 for { n := div(n, 2) } n { n := div(n,2) } {
59 let xx := mul(x, x)
60 if iszero(eq(div(xx, x), x)) { revert(0,0) }
61 let xxRound := add(xx, half)
62 if lt(xxRound, xx) { revert(0,0) }
63 x := div(xxRound, base_)
64 if mod(n,2) {
65 let zx := mul(z, x)
66 if and(iszero(iszero(x)), iszero(eq(div(zx, x), z))) { revert(0,0) }
67 let zxRound := add(zx, half)
68 if lt(zxRound, zx) { revert(0,0) }
69 z := div(zxRound, base_)
70 }
71 }
72 }
73 }

109 // --- Stability Fee Collection ---
110 function drip(bytes32 ilk) external returns (uint256 rate) {
111 require(block.timestamp >= ilks[ilk].rho, "Jug/invalid-now");
112 (, uint256 prev,,,) = vat.ilks(ilk);
113 rate = DSMath.rmul(rpow(DSMath.add(base, ilks[ilk].duty), block.timestamp - ilks[ilk].rho, DSMath.RAY), prev);
114 vat.fold(ilk, vow, int256(rate) - int256(prev));
115 ilks[ilk].rho = block.timestamp;
116 emit Drip(ilk, rate);
117 }
