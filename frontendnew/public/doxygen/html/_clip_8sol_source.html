164 // start an auction
165 // note: trusts the caller to transfer collateral into the contract
166 function kick(
167 uint256 tab, // Debt [rad]
168 uint256 lot, // Collateral [wad]
169 address usr, // Address that will receive any leftover collateral
170 address kpr // Address that will receive incentives
171 ) external auth returns (uint256 id) {
172
173 // Input validation
174 require(live == 1, "Clipper/not-live");
175 require(tab > 0, "Clipper/zero-tab");
176 require(lot > 0, "Clipper/zero-lot");
177 require(usr != address(0), "Clipper/zero-usr");
178
179 // Get starting price from spotter
180 (, uint256 rate, uint256 spot,,) = vat.ilks(ilk);
181 require(spot > 0, "Clipper/zero-spot-price");
182
183 id = kicks++;
184
185 sales[id].pos = active.length;
186 sales[id].tab = tab;
187 sales[id].lot = lot;
188 sales[id].usr = usr;
189 sales[id].tic = uint96(block.timestamp);
190
191 uint256 top = rmul(rdiv(tab, lot), buf);
192 sales[id].top = top;
193
194 active.push(id);
195
196 // incentive to kick auction
197 uint256 _tip = tip;
198 uint256 _chip = chip;
199 uint256 coin;
200 if (_tip > 0 || _chip > 0) {
201 coin = add(_tip, wmul(tab, _chip));
202 vat.move(address(vow), kpr, coin);
203 }
204
205 emit Kick(id, top, tab, lot, usr, kpr, coin);
206 }

208 // Reset an auction
209 // users can reset an auction if:
210 // - price has dropped to cusp
211 // - tail seconds have passed
212 function redo(
213 uint256 id, // id of the auction to reset
214 address kpr // Address that will receive incentives
215 ) external {
216 // Read auction data
217 address usr = sales[id].usr;
218 uint96 tic = sales[id].tic;
219 uint256 top = sales[id].top;
220
221 require(usr != address(0), "Clipper/not-running-auction");
222
223 // Check that auction needs reset
224 // and compute current price [ray]
225 (bool done,) = status(tic, top);
226 require(done, "Clipper/cannot-reset");
227
228 uint256 tab = sales[id].tab;
229 uint256 lot = sales[id].lot;
230 sales[id].tic = uint96(block.timestamp);
231
232 uint256 feedPrice = rdiv(tab, lot);
233 top = rmul(feedPrice, buf);
234 sales[id].top = top;
235
236 // incentive to redo auction
237 uint256 _tip = tip;
238 uint256 _chip = chip;
239 uint256 coin;
240 if (_tip > 0 || _chip > 0) {
241 uint256 _chost = chost;
242 if (tab >= _chost && wmul(lot, feedPrice) >= _chost) {
243 coin = add(_tip, wmul(tab, _chip));
244 vat.move(address(vow), kpr, coin);
245 }
246 }
247
248 emit Redo(id, top, tab, lot, usr, kpr, coin);
249 }

251 // Buy up to `amt` of collateral from the auction indexed by `id`
252 //
253 // Auctions will not collect more dai than their assigned `tab` (see: kick);
254 // surplus dai will be refunded to the taker.
255 // To avoid partial purchases resulting in very small leftover auctions that will
256 // never be cleared, any partial purchase must leave at least `chost` remaining `tab`.
257 function take(
258 uint256 id, // Auction id
259 uint256 amt, // Upper limit on amount of collateral to buy [wad]
260 uint256 max, // Maximum acceptable price (DAI / collateral) [ray]
261 address who, // Receiver of collateral and external call address
262 bytes calldata data // Data to pass in external call; if length 0, no call is done
263 ) external {
264
265 address usr = sales[id].usr;
266 uint96 tic = sales[id].tic;
267
268 require(live == 1, "Clipper/not-live");
269 require(usr != address(0), "Clipper/not-running-auction");
270
271 uint256 price;
272 {
273 (bool done, uint256 _price) = status(tic, sales[id].top);
274
275 // Check that auction doesn't need reset
276 require(!done, "Clipper/needs-reset");
277 price = _price;
278 }
279
280 // Ensure price is acceptable to buyer
281 require(max >= price, "Clipper/too-expensive");
282
283 uint256 lot = sales[id].lot;
284 uint256 tab = sales[id].tab;
285 uint256 owe;
286 uint256 slice;
287
288 {
289 // Purchase as much as possible, up to amt
290 slice = min(lot, amt); // slice <= lot
291
292 // DAI needed to buy a slice of this sale
293 owe = wmul(slice, price);
294
295 // Don't collect more than tab of DAI
296 if (owe > tab) {
297 // If slice == lot => auction completed => DAI needed <= tab
298 // All remaining DAI debt will be forgiven (see `heal` below)
299 owe = tab; // owe' <= owe
300 slice = wdiv(owe, price); // slice' = owe' / price <= owe / price == slice <= lot
301 } else if (owe < tab && slice < lot) {
302 // If tab > owe and slice < lot, then partial purchase at a positive (non-zero) price.
303 // In this case, we have the desired amount is being purchased.
304 // Check if remaining DAI debt is at least `chost`
305 require(sub(tab, owe) >= chost, "Clipper/no-partial-purchase");
306 }
307
308 // Adjust auction
309 sales[id].tab = sub(tab, owe);
310 sales[id].lot = sub(lot, slice);
311 vat.move(who, address(vow), owe);
312 vat.flux(ilk, address(this), who, slice);
313
314 // Removes auction from active array if no collateral left
315 if (sales[id].lot == 0) {
316 _remove(id);
317 }
318
319 emit Take(id, max, price, owe, tab, lot, who);
320 }
321
322 // Execute external call (if data.length > 0) before healing in vow
323 if (data.length > 0) {
324 ClipperCalleeFunction(who).clipperCall(msg.sender, owe, slice, data);
325 }
326
327 // Update `tab` with actual DAI collected
328 dog.digs(ilk, owe);
329
330 if (owe < tab) {
331 vow.heal(owe);
332 } else {
333 vow.heal(tab);
334 }
335 }

337 function _remove(uint256 id) internal {
338 uint256 _move = active[active.length - 1];
339 uint256 _index = sales[id].pos;
340 active[_index] = _move;
341 sales[_move].pos = _index;
342 active.pop();
343 delete sales[id];
344 }
