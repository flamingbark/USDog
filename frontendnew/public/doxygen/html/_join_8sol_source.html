23// GemJoin - Adapter for ERC20 collateral tokens (DOGE, SHIB)
24// This contract handles deposits and withdrawals of collateral tokens
25contract GemJoin {
26 using DSMath for uint256;
27
28 // --- Auth ---
29 mapping (address => uint256) public wards;
30 function rely(address usr) external auth { wards[usr] = 1; }
31 function deny(address usr) external auth { wards[usr] = 0; }
32 modifier auth { require(wards[msg.sender] == 1, "GemJoin/not-authorized"); _; }
33
34 VatLike public vat; // CDP Engine
35 bytes32 public ilk; // Collateral Type
36 GemLike public gem; // Collateral Token
37 uint256 public dec; // Token Decimals
38 uint256 public live; // Access Flag
39
40 constructor(address vat_, bytes32 ilk_, address gem_) {
41 wards[msg.sender] = 1;
42 live = 1;
43 vat = VatLike(vat_);
44 ilk = ilk_;
45 gem = GemLike(gem_);
46 dec = gem.decimals();
47 }
48
49 function cage() external auth {
50 live = 0;
51 }
52
53 function join(address usr, uint256 wad) external {
54 require(live == 1, "GemJoin/not-live");
55 require(int256(wad) >= 0, "GemJoin/overflow");
56 uint256 wad18 = wad * (10 ** (18 - dec)); // Convert token decimals to WAD (18 decimals)
57 vat.slip(ilk, usr, int256(wad18));
58 require(gem.transferFrom(msg.sender, address(this), wad), "GemJoin/failed-transfer");
59 }
60
61 function exit(address usr, uint256 wad) external {
62 require(wad <= 2 ** 255, "GemJoin/overflow");
63 uint256 wad18 = wad * (10 ** (18 - dec)); // Convert token decimals to WAD (18 decimals)
64 vat.slip(ilk, msg.sender, -int256(wad18));
65 require(gem.transfer(usr, wad), "GemJoin/failed-transfer");
66 }
67}
