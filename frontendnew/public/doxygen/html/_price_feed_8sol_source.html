95 try priceFeed.latestRoundData() returns (
96 uint80,
97 int256 price,
98 uint256,
99 uint256 updatedAt,
100 uint80
101 ) {
102 // Check if price is positive and not stale
103 bool isValid = price > 0 && block.timestamp - updatedAt <= stalePeriod;
104
105 if (isValid) {
106 // Convert to WAD (18 decimals)
107 uint256 normalizedPrice;
108 if (feedDecimals < 18) {
109 normalizedPrice = uint256(price) * (10 ** (18 - feedDecimals));

111 normalizedPrice = uint256(price) / (10 ** (feedDecimals - 18));
112 } else {
113 normalizedPrice = uint256(price);

116 }
117 } catch {
118 // If any error occurs, return invalid
119 }
120 return (bytes32(0), false);

191// Median Price Feed - Aggregates multiple price sources
192contract MedianPriceFeed {
193
194 // --- Auth ---
195 mapping (address => uint256) public wards;
196 function rely(address usr) external auth { wards[usr] = 1; }
197 function deny(address usr) external auth { wards[usr] = 0; }
198 modifier auth { require(wards[msg.sender] == 1, "MedianPriceFeed/not-authorized"); _; }
199
200 // --- Data ---
201 uint128 public val;
202 uint32 public age;
203 bytes32 public wat;
204 uint256 public bar = 1;
205
206 // Price sources
207 address[] public oracles;
208 mapping (address => uint256) public slot;
209
210 constructor(bytes32 wat_) {
211 wards[msg.sender] = 1;
212 wat = wat_;
213 }
214
215 function lift(address[] calldata a) external auth {
216 for (uint256 i = 0; i < a.length; i++) {
217 require(a[i] != address(0), "MedianPriceFeed/no-contract-0");
218 slot[a[i]] = i + 1;
219 }
220 oracles = a;
221 }
222
223 function drop(address[] calldata a) external auth {
224 for (uint256 i = 0; i < a.length; i++) {
225 slot[a[i]] = 0;
226 }
227 }
228
229 function setBar(uint256 bar_) external auth {
230 require(bar_ > 0, "MedianPriceFeed/quorum-is-zero");
231 require(bar_ % 2 != 0, "MedianPriceFeed/quorum-not-odd-number");
232 bar = bar_;
233 }
234
235 function peek() external view returns (bytes32, bool) {
236 return (bytes32(uint256(val)), val > 0);
237 }
238
239 function read() external view returns (bytes32) {
240 require(val > 0, "MedianPriceFeed/invalid-price-feed");
241 return bytes32(uint256(val));
242 }
243
244 function poke() external {
245 (val, age) = compute();
246 }
247
248 function compute() public view returns (uint128, uint32) {
249 require(oracles.length >= bar, "MedianPriceFeed/not-enough-feeds");
250
251 uint256[] memory prices = new uint256[](oracles.length);
252 uint256 validPrices = 0;
253
254 for (uint256 i = 0; i < oracles.length; i++) {
255 try PriceFeed(oracles[i]).peek() returns (bytes32 price, bool valid) {
256 if (valid) {
257 prices[validPrices] = uint256(price);
258 validPrices++;
259 }
260 } catch {
261 // Skip invalid feeds
262 }
263 }
264
265 require(validPrices >= bar, "MedianPriceFeed/not-enough-valid-feeds");
266
267 // Sort prices array (simple bubble sort for small arrays)
268 for (uint256 i = 0; i < validPrices - 1; i++) {
269 for (uint256 j = 0; j < validPrices - i - 1; j++) {
270 if (prices[j] > prices[j + 1]) {
271 uint256 temp = prices[j];
272 prices[j] = prices[j + 1];
273 prices[j + 1] = temp;
274 }
275 }
276 }
277
278 // Return median
279 uint256 medianPrice = validPrices % 2 == 0
280 ? (prices[validPrices / 2 - 1] + prices[validPrices / 2]) / 2
281 : prices[validPrices / 2];
282
283 return (uint128(medianPrice), uint32(block.timestamp));
284 }
