4// DSMath - Fixed point arithmetic library
5// Provides basic mathematical operations with different levels of precision:
6// - WAD: 18 decimal precision (10^18) - for token amounts
7// - RAY: 27 decimal precision (10^27) - for rates and ratios
8// - RAD: 45 decimal precision (10^45) - for debt calculations
9
10library DSMath {
11 uint256 constant WAD = 10 ** 18;
12 uint256 constant RAY = 10 ** 27;
13 uint256 constant RAD = 10 ** 45;

18 require((z = x + y) >= x, "ds-math-add-overflow");

27 }
