57 // --- Math ---
58 function rpow(uint256 x, uint256 n, uint256 base) internal pure returns (uint256 z) {
59 assembly {
60 switch x case 0 {switch n case 0 {z := base} default {z := 0}}
61 default {
62 switch mod(n, 2) case 0 { z := base } default { z := x }
63 let half := div(base, 2) // for rounding.
64 for { n := div(n, 2) } n { n := div(n,2) } {
65 let xx := mul(x, x)
66 if iszero(eq(div(xx, x), x)) { revert(0,0) }
67 let xxRound := add(xx, half)
68 if lt(xxRound, xx) { revert(0,0) }
69 x := div(xxRound, base)
70 if mod(n,2) {
71 let zx := mul(z, x)
72 if and(iszero(iszero(x)), iszero(eq(div(zx, x), z))) { revert(0,0) }
73 let zxRound := add(zx, half)
74 if lt(zxRound, zx) { revert(0,0) }
75 z := div(zxRound, base)
76 }
77 }
78 }
79 }

113 // --- Savings Rate Accumulation ---
114 function drip() external returns (uint256 tmp) {
115 require(block.timestamp >= rho, "Pot/invalid-now");
116 tmp = DSMath.rmul(rpow(dsr, block.timestamp - rho, DSMath.RAY), chi);
117 uint256 chi_ = sub(tmp, chi);
118 chi = tmp;
119 rho = block.timestamp;
120 vat.suck(address(vow), address(this), mul(Pie, chi_));
121 emit Drip(tmp);
122 }
123
124 // --- Savings Dai Management ---
125 function join(uint256 wad) external {
126 // Allow join any time at or after the last drip to make UX and tests deterministic
127 require(block.timestamp >= rho, "Pot/rho-not-updated");
128 pie[msg.sender] = add(pie[msg.sender], wad);
129 Pie = add(Pie, wad);
130 vat.move(msg.sender, address(this), mul(chi, wad));
131 emit Join(msg.sender, wad);
132 }
133
134 function exit(uint256 wad) external {
135 pie[msg.sender] = sub(pie[msg.sender], wad);
136 Pie = sub(Pie, wad);
137 vat.move(address(this), msg.sender, mul(chi, wad));
138 emit Exit(msg.sender, wad);
139 }
