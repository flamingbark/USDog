31// Vow - System Accountant
32// This contract manages system finances:
33// - Debt queuing system with fess()/flog() functions
34// - Surplus auctions (flap) to buy back governance tokens
35// - Deficit auctions (flop) to mint governance tokens for recapitalization
36// - Settlement mechanism for bad debt
37

115 // --- Debt Management ---
116 function fess(uint256 tab) external auth {
117 sin[block.timestamp] = add(sin[block.timestamp], tab);
118 Sin = add(Sin, tab);
119 emit Fess(block.timestamp, tab);
120 }
121
122 function flog(uint256 era) external {
123 require(add(era, wait) <= block.timestamp, "Vow/wait-not-finished");
124 Sin = sub(Sin, sin[era]);
125 emit Flog(era, sin[era]);
126 sin[era] = 0;
127 }

129 function heal(uint256 rad) external {
130 require(rad <= vat.dai(address(this)), "Vow/insufficient-dai");
131 require(rad <= sub(sub(vat.sin(address(this)), Sin), Ash), "Vow/insufficient-debt");
132 vat.heal(rad);
133 emit Heal(rad);
134 }
135
136 function kiss(uint256 rad) external {
137 require(rad <= Ash, "Vow/not-enough-ash");
138 require(rad <= vat.dai(address(this)), "Vow/insufficient-dai");
139 Ash = sub(Ash, rad);
140 vat.heal(rad);
141 emit Kiss(rad);
142 }
143
144 // --- Auction Management ---
145 function flop() external returns (uint256 id) {
146 require(sump <= sub(sub(vat.sin(address(this)), Sin), Ash), "Vow/insufficient-debt");
147 require(vat.dai(address(this)) == 0, "Vow/surplus-not-zero");
148 Ash = add(Ash, sump);
149 id = flopper.kick(address(this), dump, sump);
150 emit Flop(id, dump, sump);
151 }
152
153 function flap() external returns (uint256 id) {
154 require(vat.dai(address(this)) >= add(vat.sin(address(this)), bump), "Vow/insufficient-surplus");
155 require(sub(sub(vat.sin(address(this)), Sin), Ash) == 0, "Vow/debt-not-zero");
156 id = flapper.kick(bump, 0);
157 emit Flap(id, bump, 0);
158 }
159
160 function cage() external auth {
161 require(live == 1, "Vow/not-live");
162 live = 0;
163 Sin = 0;
164 Ash = 0;
165 flapper.cage(vat.dai(address(this)));
166 flopper.cage();
167 emit Cage();
168 }
