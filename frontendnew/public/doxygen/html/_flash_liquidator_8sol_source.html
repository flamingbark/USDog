123 // --- Main Flash Liquidation Function ---
124 function liquidateMemeCollateral(
125 address borrower,
126 address collateralVToken, // Not used in this implementation
127 address debtVToken, // Not used in this implementation
128 uint256 repayAmount,
129 address flashLoanPool
130 ) external {
131 require(!flashLoanState.inFlashLoan, "FlashLiquidator/already-in-flash-loan");
132
133 // Get collateral token from pool
134 IPancakePair pair = IPancakePair(flashLoanPool);
135 address token0 = pair.token0();
136 address token1 = pair.token1();
137
138 // Determine which token is DAI/stablecoin and which is collateral
139 address collateralToken;
140 if (token0 == address(dai)) {
141 collateralToken = token1;
142 } else if (token1 == address(dai)) {
143 collateralToken = token0;
144 } else {
145 // Neither token is DAI, this is a collateral-WBNB pair
146 // We'll need to determine collateral based on supported list
147 if (supportedCollateral[token0]) {
148 collateralToken = token0;
149 } else if (supportedCollateral[token1]) {
150 collateralToken = token1;
151 } else {
152 revert("FlashLiquidator/unsupported-collateral");
153 }
154 }
155
156 require(supportedCollateral[collateralToken], "FlashLiquidator/unsupported-collateral");
157
158 // Set up flash loan state
159 flashLoanState = FlashLoanData({
160 borrower: borrower,
161 collateralToken: collateralToken,
162 collateralJoin: collateralJoins[collateralToken],
163 repayAmount: repayAmount,
164 expectedCollateral: 0, // Will be calculated
165 inFlashLoan: true
166 });
167
168 // Calculate how much DAI we need for the liquidation
169 uint256 flashAmount = repayAmount / 1e18; // Convert from rad to wad
170
171 // Initiate flash loan by calling swap with data
172 bytes memory data = abi.encode(borrower, collateralToken, repayAmount);
173
174 // Determine swap amounts based on pair configuration
175 if (token0 == address(dai)) {
176 pair.swap(flashAmount, 0, address(this), data);
177 } else if (token1 == address(dai)) {
178 pair.swap(0, flashAmount, address(this), data);
179 } else {
180 // For collateral-WBNB pairs, we need to flash WBNB first
181 // then swap to DAI in the callback
182 uint256 wbnbAmount = getWBNBForDAI(flashAmount);
183 if (token0 == WBNB) {
184 pair.swap(wbnbAmount, 0, address(this), data);
185 } else {
186 pair.swap(0, wbnbAmount, address(this), data);
187 }
188 }
189 }

191 // --- PancakeSwap Flash Loan Callback ---
192 function pancakeCall(address sender, uint256 amount0, uint256 amount1, bytes calldata data) external {
193 require(flashLoanState.inFlashLoan, "FlashLiquidator/not-in-flash-loan");
194
195 // Verify caller is a legitimate PancakeSwap pair
196 address token0 = IPancakePair(msg.sender).token0();
197 address token1 = IPancakePair(msg.sender).token1();
198 require(msg.sender == pairFor(token0, token1), "FlashLiquidator/invalid-pair");
199
200 (address borrower, address collateralToken, uint256 repayAmount) = abi.decode(data, (address, address, uint256));
201
202 uint256 flashAmount = amount0 > 0 ? amount0 : amount1;
203
204 // If we flashed WBNB, swap it to DAI first
205 if ((amount0 > 0 && token0 == WBNB) || (amount1 > 0 && token1 == WBNB)) {
206 // Swap WBNB to DAI
207 address[] memory path = new address[](2);
208 path[0] = WBNB;
209 path[1] = address(dai);
210
211 IERC20(WBNB).approve(address(router), flashAmount);
212 uint256[] memory amounts = router.swapExactTokensForTokens(
213 flashAmount,
214 0, // Accept any amount of DAI
215 path,
216 address(this),
217 block.timestamp + 300
218 );
219 flashAmount = amounts[1]; // Update to DAI amount received
220 }
221
222 // Convert DAI to internal DAI for liquidation
223 dai.approve(address(daiJoin), flashAmount);
224 daiJoin.join(address(this), flashAmount);
225
226 // Perform the liquidation by calling clipper
227 // This will trigger clipperCall which handles the actual liquidation
228 executeClipperLiquidation(borrower, collateralToken, repayAmount);
229
230 // Calculate repayment with fee (0.3% PancakeSwap fee)
231 uint256 repaymentAmount = flashAmount * 1003 / 1000;
232
233 // Ensure we have enough DAI to repay
234 uint256 daiBalance = dai.balanceOf(address(this));
235 require(daiBalance >= repaymentAmount, "FlashLiquidator/insufficient-repayment");
236
237 // Repay the flash loan
238 if (token0 == address(dai)) {
239 dai.transfer(msg.sender, repaymentAmount);
240 } else if (token1 == address(dai)) {
241 dai.transfer(msg.sender, repaymentAmount);
242 } else {
243 // Need to swap DAI back to WBNB for repayment
244 address[] memory path = new address[](2);
245 path[0] = address(dai);
246 path[1] = WBNB;
247
248 dai.approve(address(router), repaymentAmount);
249 router.swapExactTokensForTokens(
250 repaymentAmount,
251 0,
252 path,
253 msg.sender,
254 block.timestamp + 300
255 );
256 }
257
258 // Calculate and emit profit
259 uint256 finalBalance = dai.balanceOf(address(this));
260 uint256 profit = finalBalance > daiBalance - repaymentAmount ?
261 finalBalance - (daiBalance - repaymentAmount) : 0;
262
263 emit FlashLiquidation(borrower, collateralToken, repayAmount, flashLoanState.expectedCollateral, profit);
264
265 // Clear flash loan state
266 delete flashLoanState;
267 }

269 // --- Clipper Callback (called during auction participation) ---
270 function clipperCall(address sender, uint256 owe, uint256 slice, bytes calldata data) external {
271 // This is called by the Clipper when we participate in an auction
272 // owe: DAI amount we owe for the collateral
273 // slice: collateral amount we're receiving
274
275 require(flashLoanState.inFlashLoan, "FlashLiquidator/not-in-flash-loan");
276
277 // Extract collateral from the contract and convert to external tokens
278 GemJoinLike collateralJoin = GemJoinLike(flashLoanState.collateralJoin);
279 collateralJoin.exit(address(this), slice);
280
281 // Swap collateral to DAI via PancakeSwap
282 IERC20 collateralToken = IERC20(flashLoanState.collateralToken);
283 uint256 collateralBalance = collateralToken.balanceOf(address(this));
284
285 if (collateralBalance > 0) {
286 // Determine swap path
287 address[] memory path;
288 if (flashLoanState.collateralToken == WBNB) {
289 path = new address[](2);
290 path[0] = flashLoanState.collateralToken;
291 path[1] = address(dai);
292 } else {
293 path = new address[](3);
294 path[0] = flashLoanState.collateralToken;
295 path[1] = WBNB;
296 path[2] = address(dai);
297 }
298
299 collateralToken.approve(address(router), collateralBalance);
300 router.swapExactTokensForTokens(
301 collateralBalance,
302 0, // Accept any amount of DAI (market rate)
303 path,
304 address(this),
305 block.timestamp + 300
306 );
307 }
308
309 flashLoanState.expectedCollateral = slice;
310 }
