120 // --- CDP Liquidation ---
121 function bark(bytes32 ilk, address urn, address kpr) external returns (uint256 id) {
122 require(live == 1, "Dog/not-live");
123
124 (uint256 ink, uint256 art) = vat.urns(ilk, urn);
125 (, uint256 rate, uint256 spot, uint256 line, uint256 dust) = vat.ilks(ilk);
126 uint256 tab = mul(art, rate);
127
128 require(spot > 0 && mul(ink, spot) < tab, "Dog/not-unsafe");
129
130 Ilk memory milk = ilks[ilk];
131 uint256 room;
132 {
133 uint256 dart;
134 uint256 dtab;
135 uint256 dink;
136
137 room = min(milk.hole, sub(Hole, Dirt));
138 dart = min(art, mul(room, DSMath.WAD) / rate / milk.chop);
139 dink = min(ink, mul(ink, dart) / art);
140 dtab = mul(dart, rate);
141
142 require(dtab > 0, "Dog/null-auction");
143 require(dtab >= dust || (dtab == tab && tab > 0), "Dog/dusty-auction");
144
145 vat.grab(ilk, urn, address(this), address(vow), -int256(dink), -int256(dart));
146
147 uint256 due = mul(dtab, milk.chop) / DSMath.WAD;
148 vow.fess(sub(due, dtab));
149
150 {
151 Dirt = add(Dirt, due);
152 ilks[ilk].dirt = add(milk.dirt, due);
153 }
154
155 id = milk.clip.kick(tab, due, urn, kpr);
156
157 emit Bark(ilk, urn, dink, dart, due, address(milk.clip), id);
158 }
159 }
