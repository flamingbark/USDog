58 // Execute multiple calls in a single transaction
59 function aggregate(Call[] memory calls) public payable returns (uint256 blockNumber, bytes[] memory returnData) {
60 require(live == 1, "Multicall/not-live");
61 blockNumber = block.number;
62 returnData = new bytes[](calls.length);
63
64 for (uint256 i = 0; i < calls.length; i++) {
65 (bool success, bytes memory ret) = calls[i].target.call(calls[i].callData);
66 require(success, "Multicall/call-failed");
67 returnData[i] = ret;
68 }
69 }
70
71 // Execute multiple calls, allowing failures
72 function tryAggregate(bool requireSuccess, Call[] memory calls) public payable returns (Result[] memory returnData) {
73 require(live == 1, "Multicall/not-live");
74 returnData = new Result[](calls.length);
75
76 for (uint256 i = 0; i < calls.length; i++) {
77 (bool success, bytes memory ret) = calls[i].target.call(calls[i].callData);
78 if (requireSuccess) {
79 require(success, "Multicall/call-failed");
80 }
81 returnData[i] = Result(success, ret);
82 }
83 }

85 // Execute multiple calls and return block hash and number
86 function blockAndAggregate(Call[] memory calls) public payable returns (uint256 blockNumber, bytes32 blockHash, Result[] memory returnData) {
87 require(live == 1, "Multicall/not-live");
88 (blockNumber, blockHash, returnData) = (block.number, blockhash(block.number), tryAggregate(true, calls));
