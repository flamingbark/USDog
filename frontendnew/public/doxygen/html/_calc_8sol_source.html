6// LinearDecrease - Linear price decrease calculator
7// Price decreases linearly over time for Dutch auctions
8
9contract LinearDecrease {
10 using DSMath for uint256;
11
12 // --- Auth ---
13 mapping (address => uint256) public wards;
14 function rely(address usr) external auth { wards[usr] = 1; }
15 function deny(address usr) external auth { wards[usr] = 0; }
16 modifier auth { require(wards[msg.sender] == 1, "LinearDecrease/not-authorized"); _; }
17
18 // --- Data ---
19 uint256 public tau; // Seconds after auction start when the price reaches zero [seconds]
20
21 constructor() {
22 wards[msg.sender] = 1;
23 tau = 3600; // Default 1 hour
24 }
25
26 // --- Administration ---
27 function file(bytes32 what, uint256 data) external auth {
28 if (what == "tau") tau = data;
29 else revert("LinearDecrease/file-unrecognized-param");
30 }
31
32 // --- Math ---
33 function add(uint256 x, uint256 y) internal pure returns (uint256 z) {
34 require((z = x + y) >= x);
35 }
36 function sub(uint256 x, uint256 y) internal pure returns (uint256 z) {
37 require((z = x - y) <= x);
38 }
39 function mul(uint256 x, uint256 y) internal pure returns (uint256 z) {
40 require(y == 0 || (z = x * y) / y == x);
41 }
42
43 // --- Pricing ---
44 function price(uint256 top, uint256 dur) external view returns (uint256) {
45 if (dur >= tau) return 0;
46 return sub(top, mul(top, dur) / tau);
47 }

50// StairstepExponentialDecrease - Exponential price decrease in steps
51// Price decreases exponentially in discrete steps for more predictable auctions
52
53contract StairstepExponentialDecrease {
54 using DSMath for uint256;
55
56 // --- Auth ---
57 mapping (address => uint256) public wards;
58 function rely(address usr) external auth { wards[usr] = 1; }
59 function deny(address usr) external auth { wards[usr] = 0; }
60 modifier auth { require(wards[msg.sender] == 1, "StairstepExponentialDecrease/not-authorized"); _; }
61
62 // --- Data ---
63 uint256 public step; // Length of time between price drops [seconds]
64 uint256 public cut; // Per-step multiplicative factor [ray]
65
66 constructor() {
67 wards[msg.sender] = 1;
68 step = 300; // Default 5 minutes
69 cut = 995000000000000000000000000; // Default 0.5% reduction per step
70 }
71
72 // --- Administration ---
73 function file(bytes32 what, uint256 data) external auth {
74 if (what == "cut") {
75 require(data < DSMath.RAY, "StairstepExponentialDecrease/cut-too-high");
76 cut = data;
77 }
78 else if (what == "step") {
79 require(data > 0, "StairstepExponentialDecrease/step-too-low");
80 step = data;
81 }
82 else revert("StairstepExponentialDecrease/file-unrecognized-param");
83 }
84
85 // --- Math ---
86 uint256 constant RAY = 10 ** 27;
87
88 function rmul(uint256 x, uint256 y) internal pure returns (uint256 z) {
89 z = DSMath.add(DSMath.mul(x, y), RAY / 2) / RAY;
90 }
91
92 function rpow(uint256 x, uint256 n) internal pure returns (uint256 z) {
93 z = n % 2 != 0 ? x : RAY;
94
95 for (n /= 2; n != 0; n /= 2) {
96 x = rmul(x, x);
97 if (n % 2 != 0) {
98 z = rmul(z, x);
99 }
100 }
101 }
102
103 // --- Pricing ---
104 function price(uint256 top, uint256 dur) external view returns (uint256) {
105 if (step == 0) return 0;
106
107 uint256 elapsed = dur / step;
108 if (elapsed == 0) return top;
109
110 return rmul(top, rpow(cut, elapsed));
111 }
112}
