106 function add(uint256 x, uint256 y) internal pure returns (uint256 z) {
107 z = x + y;
108 require(z >= x);

110 function sub(uint256 x, uint256 y) internal pure returns (uint256 z) {
111 require((z = x - y) <= x);

113 function mul(uint256 x, uint256 y) internal pure returns (uint256 z) {
114 require(y == 0 || (z = x * y) / y == x);

116 function min(uint256 x, uint256 y) internal pure returns (uint256 z) {

176 function skim(bytes32 ilk, address urn) external {
177 require(tag[ilk] != 0, "End/tag-not-set");
178 (uint256 ink, uint256 art) = vat.urns(ilk, urn);
179 (, uint256 rate,,,) = vat.ilks(ilk);
180 uint256 owe = rmul(rmul(art, rate), tag[ilk]);
181 uint256 wad = min(ink, owe);
182 gap[ilk] = add(gap[ilk], sub(owe, wad));
183
184 vat.grab(ilk, urn, address(this), address(vow), -int256(wad), -int256(art));
185 emit Skim(ilk, urn, wad, art);
186 }

195 function thaw() external {

204 function flow(bytes32 ilk) external {
205 require(debt != 0, "End/not-processing");
206 (, uint256 rate,,,) = vat.ilks(ilk);
207 require(fix[ilk] == 0, "End/fix-set");
208
209 Art[ilk] = rmul(Art[ilk], rate);
210 fix[ilk] = wdiv(sub(wdiv(Art[ilk], debt), gap[ilk]), Art[ilk]);
211 emit Flow(ilk);
212 }
213
214 function pack(uint256 wad) external {
215 require(debt != 0, "End/not-processing");
216 vat.move(msg.sender, address(vow), mul(wad, RAY));
217 bag[msg.sender] = add(bag[msg.sender], wad);
218 emit Pack(msg.sender, wad);
219 }
220
221 function cash(bytes32 ilk, uint256 wad) external {
222 require(fix[ilk] != 0, "End/fix-not-set");
223 vat.flux(ilk, address(this), msg.sender, rmul(wad, fix[ilk]));
224 out[ilk][msg.sender] = add(out[ilk][msg.sender], wad);
225 require(out[ilk][msg.sender] <= bag[msg.sender], "End/insufficient-bag-balance");
226 emit Cash(ilk, msg.sender, wad);
227 }
